Reverse the String

 var Reverse_String = (str) => 
{
  const tempArr = []
  let reversedStr = ''
  for(let i=str.length-1; i >= 0; i--){
    tempArr.push(str.charAt(i))
  }
  
  for(let i=0; i < tempArr.length; i++){
    reversedStr += tempArr[i]
  }
  
  return reversedStr
};

String Replace

 var Replace = (S, pattern , text) => 
{
  let ans = ""
  let index = S.search(pattern);
  for(let i=0; i < index; i++){
    ans += S[i]
  }
  ans += text;
  for(let i=index+pattern.length; i < S.length; i++){
    ans += S[i]
  }
  return ans
};
